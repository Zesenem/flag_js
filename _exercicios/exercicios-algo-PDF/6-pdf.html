// Installments calculator with interest

function runJS() {
  let valor = Number(prompt("Digite o valor da compra em R$:"));
  let taxa = Number(prompt("Digite a taxa de juros em %:"));
  let tempo = Number(prompt("Digite o tempo de pagamento em meses:"));

  let parcelas = valor + (valor * (taxa / 100) * tempo);

  return `
    
Valor da compra: R$ ${valor.toFixed(2)}

    
Taxa de juros: ${taxa.toFixed(2)}%

    
Tempo de pagamento: ${tempo} meses

    
Total a pagar: R$ ${parcelas.toFixed(2)}

  `;
}
