JavaScript Implementation

// Dollar to Euro Converter

function runJS() {
  let euro = Number(prompt("Digite o valor em Euros:"));
  let quote = Number(prompt("Digite a cotação do Dólar:"));

  let dollar = euro * quote;

  return `
Valor em Dólares: $${dollar.toFixed(2)}
`;
}
