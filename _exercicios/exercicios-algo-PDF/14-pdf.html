Student Pass or Fail

Portugol Pseudocode

inicio
  real a, b, c, d, media

  escrever "digite a primeira nota do aluno "
  ler a

  escrever "digite a segunda nota do aluno "
  ler b

  escrever "digite a terceira nota do aluno "
  ler c

  escrever "digite a quarta nota do aluno "
  ler d

  media <- (a + b + c + d) / 4 

  se media >= 5 entao
    escrever "o aluno foi aprovado com a media de ", media
  senao 
    escrever "o aluno chumbou com a media de ", media
  fimSe

fim

JavaScript Implementation

// Student Pass or Fail Calculator

function runJS() {
  let num1 = Number(prompt("Digite a nota 1"));
  let num2 = Number(prompt("Digite a nota 2"));
  let num3 = Number(prompt("Digite a nota 3"));
  let num4 = Number(prompt("Digite a nota 4"));

  let sum = num1 + num2 + num3 + num4;
  let average = sum / 4;

  if (average >= 5) {
    return `
Aprovado: ${average}
`;
  } else {
    return `
Reprovado: ${average}
`;
  }
}
