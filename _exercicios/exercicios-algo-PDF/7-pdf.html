// Swap values of A and B using a temporary variable C

function runJS() {
  let a = Number(prompt("Digite o valor de A:"));
  let b = Number(prompt("Digite o valor de B:"));

  let c = a;
  a = b;
  b = c;

  return `
    
Valor de A após troca: ${a}

    
Valor de B após troca: ${b}

  `;
}
