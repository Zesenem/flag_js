// Volume of a Cylinder

function runJS() {
  let altura = Number(prompt("Digite a altura:"));
  let raio = Number(prompt("Digite o raio:"));

  let volume = Math.PI * Math.pow(raio, 2) * altura;

  return `
Volume: ${volume.toFixed()}
`;
}
