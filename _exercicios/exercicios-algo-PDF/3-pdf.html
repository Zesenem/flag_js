// Fahrenheit to Celsius Converter

function runJS() {
  let temp = Number(prompt("Digite a temperatura em Fahrenheit:"));
  let celsius = (temp - 32) * 5 / 9;
  return `
Temperatura em Celsius: ${celsius.toFixed()}
`;
}
