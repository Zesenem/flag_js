  // Trapeze Area Example

  function runJS() {
    let base1 = Number(prompt("Digite a base 1:"));
    let base2 = Number(prompt("Digite a base 2:"));
    let altura = Number(prompt("Digite a altura:"));
    
    let area = ((base1 + base2) * altura) / 2;
    
    return `
Area of the Trapeze: ${area}
`;
  }
