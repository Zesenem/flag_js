Portugol Pseudocode

inicio
    real a, b, c, media
    
    escrever "digite o valor A "
    ler a
    
    escrever "digite o valor B "
    ler b
    
    media <- (a + b) / 2 
    
    se media >= 7 entao
        escrever "o aluno foi aprovado com a media: ", media
    senao
        escrever "digite o valor C da nota de exame: "
        ler c
        
        media <- (a + b + c) / 3
        
        se media >= 7 entao
            escrever "o aluno foi aprovado com exame com a media: ", media
        senao
            escrever "o aluno foi reprovado com exame com a media: ", media
        fimSe
    fimSe
fim

JavaScript Implementation

// Student Pass or Fail with Exam Calculator

function runJS() {
  let num1 = Number(prompt("Digite a nota 1"));
  let num2 = Number(prompt("Digite a nota 2"));

  let sum = num1 + num2;
  let average = sum / 2;

  if (average >= 7) {
    return `
Aprovado: ${average}
`;
  } else {
    let exam = Number(prompt("Digite a nota do exame"));
    let final = (average + exam) / 2;
    if (final >= 7) {
      return `
Aprovado com Exame: ${final}
`;
    } else {
      return `
Reprovado: ${final}
`;
    }
  }
}
