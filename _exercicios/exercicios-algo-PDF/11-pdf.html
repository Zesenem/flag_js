Portugol Pseudocode

inicio
  real a, b, c, total

  escrever "digite valor a: "
  ler a

  escrever "digite valor b: "
  ler b

  escrever "digite valor c: "

  total ← a^2 + b^2 + c^2

JavaScript Implementation

// Square Sum Calculator

function runJS() {
  let a = Number(prompt("Digite o valor de A"));
  let b = Number(prompt("Digite o valor de B"));
  let c = Number(prompt("Digite o valor de C"));

  let total = (a * a) + (b * b) + (c * c);

  return `
Resultado: ${total}
`;
}
