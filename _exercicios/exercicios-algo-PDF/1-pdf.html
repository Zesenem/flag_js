// JavaScript Square Sum Example
function runJS() {
  let num1 = Number(prompt("Digite o primeiro número:"));
  let num2 = Number(prompt("Digite o segundo número:"));
  let sum = (num1 * num1) + (num2 * num2);
  return `
Square Sum: ${sum}
`;
}
