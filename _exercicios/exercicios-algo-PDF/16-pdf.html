JavaScript Implementation

// Rising Order Calculator

function runJS() {
  let num1 = Number(prompt("Digite o primeiro número"));
  let num2 = Number(prompt("Digite o segundo número"));
  let num3 = Number(prompt("Digite o terceiro número"));

  if (num1 < num2 && num1 < num3) {
    if (num2 < num3) {
      return `
Ordem Crescente: ${num1}, ${num2}, ${num3}
`;
    } else {
      return `
Ordem Crescente: ${num1}, ${num3}, ${num2}
`;
    }
  } else if (num2 < num1 && num2 < num3) {
    if (num1 < num3) {
      return `
Ordem Crescente: ${num2}, ${num1}, ${num3}
`;
    } else {
      return `
Ordem Crescente: ${num2}, ${num3}, ${num1}
`;
    }
  } else if (num3 < num1 && num3 < num2) {
    if (num1 < num2) {
      return `
Ordem Crescente: ${num3}, ${num1}, ${num2}
`;
    } else {
      return `
Ordem Crescente: ${num3}, ${num2}, ${num1}
`;
    }
  }
}
