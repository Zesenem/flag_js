JavaScript Implementation

// Difference of Higher and Lower Calculator

function runJS() {
  let a = Number(prompt("Digite o valor de A"));
  let b = Number(prompt("Digite o valor de B"));
  
  let difference = 0;

  if (a > b) {
    difference = a - b;
  } else {
    difference = b - a;
  }

  return `
Resultado: ${difference}
`;
}
