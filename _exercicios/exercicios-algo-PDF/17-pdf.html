JavaScript Implementation

// Divisible by 2 and 3

function checkDivisible() {
  let num1 = Number(prompt("Digite o primeiro número"));
  let num2 = Number(prompt("Digite o segundo número"));
  let num3 = Number(prompt("Digite o terceiro número"));
  let num4 = Number(prompt("Digite o quarto número"));
  let output = document.getElementById("output");

  if (num1 % 2 == 0 && num1 % 3 == 0) {
    output.innerHTML += `
Divisível por 2 e 3: ${num1}
`;
  } else {
    output.innerHTML += `
Não é divisível por 2 e 3: ${num1}
`;
  }

  if (num2 % 2 == 0 && num2 % 3 == 0) {
    output.innerHTML += `
Divisível por 2 e 3: ${num2}
`;
  } else {
    output.innerHTML += `
Não é divisível por 2 e 3: ${num2}
`;
  }

  if (num3 % 2 == 0 && num3 % 3 == 0) {
    output.innerHTML += `
Divisível por 2 e 3: ${num3}
`;
  } else {
    output.innerHTML += `
Não é divisível por 2 e 3: ${num3}
`;
  }

  if (num4 % 2 == 0 && num4 % 3 == 0) {
    output.innerHTML += `
Divisível por 2 e 3: ${num4}
`;
  } else {
    output.innerHTML += `
Não é divisível por 2 e 3: ${num4}
`;
  }
}
