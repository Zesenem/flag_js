// Difference of Squares

function runJS() {
  let a = Number(prompt("Digite o primeiro número:"));
  let b = Number(prompt("Digite o segundo número:"));

  let difference = (a + b) * (a - b);

  return `
Diferença dos quadrados: ${difference}
`;
}
