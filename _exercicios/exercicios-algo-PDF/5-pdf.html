// Gasoline consumption calculator

function runJS() {
  let tempo = Number(prompt("Digite o tempo de viagem:"));
  let velocidade = Number(prompt("Digite a velocidade média:"));

  let distancia = tempo * velocidade;
  let gasolina = distancia / 12;

  return `
Gasolina: ${gasolina.toFixed(2)} L
`;
}
